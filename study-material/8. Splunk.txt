# Requests served by each host

------------------------------------------------------------------------------------------
# Total number of API hits
------------------------------------------------------------------------------------------
index=pim_core sourcetype=pim_log REQUEST_URI pimservers | timechart span=1d count as REQUEST_URI

------------------------------------------------------------------------------------------
# Number of hits per API
------------------------------------------------------------------------------------------
index=pim_core sourcetype=pim_log 
	LoggingAspect ("ENTERED_METHOD=USIController.processProvisionRequest()" OR "ENTERED_METHOD=USIController.processPersonProfile()" ) 
	| replace USIController.processProvisionRequest() WITH user(POST), USIController.processPersonProfile() WITH Profile(POST) IN ENTERED_METHOD 
	| chart count by ENTERED_METHOD

------------------------------------------------------------------------------------------
# API performance
------------------------------------------------------------------------------------------
index=pim_core sourcetype=pim_log LoggingAspect EXITED_METHOD="*Controller*" RESPONSE_TIME | chart avg(RESPONSE_TIME) by EXITED_METHOD

# Exceptions occurred in each host servers

index=pim_core sourcetype=pim_log EXCEPTION_MESSAGE LoggingAspect | timechart span=24h count by host
